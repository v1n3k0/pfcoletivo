drop database unifunda;
Create database unifunda;
use unifunda;

Create Table usuario (
  login varchar(20),
  senha varchar(20),
    nome varchar(20),
    cpf bigint NOT NULL PRIMARY KEY,
    pais varchar(20),
    cidade varchar(50),
    estado varchar(50),
    endereco varchar(200),
    data_n date,
    email varchar(40),
    tipo varchar(20),
    categoria varchar(50)
);
Create Table criterio (
  cod_cri int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    nome_cat varchar(30)
);

Create Table categoria (
  cod_cat int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    cod_cri_fk int NOT NULL,
    nome_cat varchar(30),
    CONSTRAINT cod_cri_fk FOREIGN KEY (cod_cri_fk)REFERENCES criterio(cod_cri)
    
);

Create Table projeto (
  cod_p int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    nome varchar(20),
    cod_cat_fk int,
  cod_cri_fk int,
    duracao int,
    valor float,
    statu varchar(20),
    descricao varchar(255),
    nota int,
    video varchar(255),
    imagem varchar(200),
    CONSTRAINT cod_cr_fk FOREIGN KEY (cod_cri_fk)REFERENCES criterio(cod_cri),
    CONSTRAINT cod_ca_fk FOREIGN KEY (cod_cat_fk)REFERENCES categoria(cod_cat)
    );
    
Create Table financiar (
  cpf_fk bigint NOT NULL,
  cod_p_fk int NOT NULL,
    valor_doado float,
    form_pag varchar(30),
    data_fin date,
    CONSTRAINT cpf_fk FOREIGN KEY (cpf_fk)REFERENCES usuario(cpf),
    CONSTRAINT cod_p_fk FOREIGN KEY (cod_p_fk)REFERENCES projeto(cod_p),
    PRIMARY KEY  (cpf_fk,cod_p_fk)
);
